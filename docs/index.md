```typescript
// purely functional schema definition
import { defineSchema, s } from "mondel";

export const userSchema = defineSchema("users", {
  timestamps: true, // adds createdAt, updatedAt
  fields: {
    // _id is implicit - auto-generated by MongoDB
    email: s.string().required().unique(),
    role: s.enum(["ADMIN", "USER"]).default("USER"),
    metadata: s.object({
      active: s.boolean(),
    }),
  },
});
```

  </div>

  <div v-show="activeTab === 'client'" class="code-block">

```typescript
// optimized for serverless edge environments
import { createClient } from "mondel";

// lightweight factory, no persistent connection
const connect = createClient({
  serverless: true,
  schemas: [userSchema],
  validation: "strict", // Zod validation enabled
});

// use inside your handler
const db = await connect(env.MONGO_URI);
```

  </div>

```typescript
// fully typed results based on schema
const user = await db.users.findOne({
  email: "alice@example.com",
});

// types are inferred automatically!
console.log(user.role); // "ADMIN" | "USER"
console.log(user.metadata.active); // boolean

// typescript error: property doesn't exist
console.log(user.password);
```
